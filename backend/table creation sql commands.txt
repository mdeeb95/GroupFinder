CREATE TABLE members(memberID TEXT PRIMARY KEY NOT NULL, name TEXT NOT NULL, major TEXT NOT NULL, year TEXT NOT NULL, location TEXT NOT NULL, info TEXT NOT NULL, password TEXT NOT NULL);

CREATE TABLE classes(classID INT PRIMARY KEY NOT NULL, name TEXT NOT NULL, leader TEXT NOT NULL, type TEXT NOT NULL);

CREATE TABLE class_members (classID INT NOT NULL, memberID INT NOT NULL, PRIMARY KEY(classID, memberID));
